50 static inline lbm_exp_kind lbm_exp_kind_of(lbm_value exp) {
51 
52  switch (lbm_type_of(exp)) {
53  case LBM_VAL_TYPE_SYMBOL:
54  if (!lbm_is_special(exp))
55  return EXP_VARIABLE;
56  // fall through
57  case LBM_PTR_TYPE_BOXED_F:
58  case LBM_PTR_TYPE_BOXED_U:
59  case LBM_PTR_TYPE_BOXED_I:
60  case LBM_VAL_TYPE_I:
61  case LBM_VAL_TYPE_U:
62  case LBM_VAL_TYPE_CHAR:
63  case LBM_PTR_TYPE_ARRAY:
64  return EXP_SELF_EVALUATING;
65  case LBM_PTR_TYPE_CONS: {
66  lbm_value head = lbm_car(exp);
67  if (lbm_type_of(head) == LBM_VAL_TYPE_SYMBOL) {
68  lbm_uint sym_id = lbm_dec_sym(head);
69  switch(sym_id){
70  case SYM_AND: return EXP_AND;
71  case SYM_OR: return EXP_OR;
72  case SYM_QUOTE: return EXP_QUOTED;
73  case SYM_DEFINE: return EXP_DEFINE;
74  case SYM_PROGN: return EXP_PROGN;
75  case SYM_LAMBDA: return EXP_LAMBDA;
76  case SYM_IF: return EXP_IF;
77  case SYM_LET: return EXP_LET;
78  default: break;
79  }
80  } // end if symbol
81 
82  if (lbm_type_of(lbm_cdr(exp)) == LBM_VAL_TYPE_SYMBOL &&
83  lbm_dec_sym(lbm_cdr(exp)) == SYM_NIL) {
84  return EXP_NO_ARGS;
85  } else {
86  return EXP_APPLICATION;
87  }
88  } // end case PTR_TYPE_CONS:
89  }
90  return EXP_KIND_ERROR;
